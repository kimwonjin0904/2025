파일선택을 눌렀을때 파일
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>이미지 업로드</title>
    <script src="https://code.jquery.com/jquery-3.7.1.js"></script>
    <style>
        body{text-align: center;}
        .input-label{padding:8px;
                        border: 1px solid #ccc; cursor: pointer;}
        #preview{max-width: 300px; margin-top: 10px; border-radius: 20px;}

    </style>

    <script>
        $(document).ready(function(){
            $("#imgFile").change(function(e){
                let file = e.target.files[0];
                if(file && file.type.match("image.*")){
                    let reader = new FileReader();
                    reader.onload = function(event){
                        console.log(event.target.result);
                        $("#preview").attr("src",event.target.result);
                        $("#preview").show();
                    }
                    //이미지 파일을 -> base64
                    //브라우저에서 출력 할 수 있는 base64로 변환
                    //이미지 파일 크기 보다 33% 더커짐
                    reader.readAsDataURL(file);
                    //텍스트는 readAstext
                }else{
                    alert("유요한 이미지 파일을 선택하세요!!");
                }
            });
        });
    </script>
</head>


<body>
    <h3>이미지 업로드</h3>
    <div class="container">
        <label for="imgFile" class="input-label">파일 선택</label>
        <input type="file" 
                 id="imgFile" accept="image/*"  style="display: none;"> <!--accept="image/* 은 파일을 불러오기 위한 함수"-->
    </div>

    <img src="" id="preview" style="display: none;" alt="">
</body>
